Л1. Вычислить s=1^2+3^2+5^2+...+n^2, n - нечетное
число

var
	i, n: integer;
	s: real;
begin
	readln(n);
	s := 0;
	for i := 1 to n do
		if i mod 2 <> 0 then
			s := s + i * i;
	writeln('s=', s);
end.
